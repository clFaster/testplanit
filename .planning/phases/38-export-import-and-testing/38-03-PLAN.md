      const llmId = await api.createLlmIntegration(`E2E Override ${Date.now()}`);
      await api.linkLlmToProject(projectId, llmId);

      await page.goto(`/en-US/projects/settings/${projectId}/ai-models`);
      await page.waitForLoadState("networkidle");

      // Find the Feature Overrides table
      // The FeatureOverrides component renders a Table inside a Card with title matching
      // translation key "projects.settings.aiModels.featureOverrides.title"
      const featureOverridesCard = page.locator("table").last();

      // Find the row for "Test Case Generation"
      const row = featureOverridesCard.locator("tr", { hasText: "Test Case Generation" });

      // Click the override Select in that row (the second cell has the Select)
      const selectTrigger = row.locator('button[role="combobox"]');
      await selectTrigger.click();

      // Select the created integration
      const option = page.getByRole("option", { name: /E2E Override/ });
      await option.click();

      // Wait for the save toast
      await page.waitForTimeout(1000);

      // Verify the "Effective LLM" column shows the integration name
      await expect(row).toContainText(/E2E Override/);

      // Verify the "Source" column shows "Project Override" badge
